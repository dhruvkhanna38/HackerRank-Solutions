SELECT H.HACKER_ID, NAME, COUNT(1)
FROM HACKERS H LEFT JOIN CHALLENGES C ON H.HACKER_ID = C.HACKER_ID
GROUP BY H.HACKER_ID, H.NAME
HAVING COUNT(1) = (
SELECT TOP 1 COUNT(1)
FROM HACKERS H LEFT JOIN CHALLENGES C ON H.HACKER_ID = C.HACKER_ID
GROUP BY H.HACKER_ID, H.NAME
ORDER BY 1 DESC)
OR COUNT(1) IN (SELECT CHALL FROM (
    SELECT COUNT(1) AS CHALL
FROM HACKERS H LEFT JOIN CHALLENGES C ON H.HACKER_ID = C.HACKER_ID
GROUP BY H.HACKER_ID, H.NAME
) T GROUP BY CHALL HAVING COUNT(*) = 1 
)
ORDER BY 3 DESC, HACKER_ID;
